#!groovy

// Required: https://plugins.jenkins.io/pipeline-utility-steps

node {
  def nodes = nodesByLabel label:"blender"
  def nodesSorted = nodes.sort().toList()
  def nodeTasks = [:]

  for (int i = 0; i < nodesSorted.size(); ++i) {
    def nodeName = nodesSorted[i]

    nodeTasks[nodeName] = {
      node(nodeName) {
        stage(nodeName) {
          checkout scm
          sh 'env | sort'
          sh './work.sh'
        }
      }
    }
  }

  parallel nodeTasks
}
