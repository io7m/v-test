#!groovy

// Required: https://plugins.jenkins.io/pipeline-utility-steps

pipeline {

  agent none

  stages {
    stage('BuildAndTest') {
      parallel {
        stage('x') {
          agent { label 'blender' }
          steps {
            sh 'env | sort'

            script {
              def nodes = nodesByLabel label:"blender"
              for (int i = 0; i < nodes.size(); ++i) {
                echo "Nodes: ${nodes[i]}"
              }
            }
          }
        }
      }
    }
  }
}
